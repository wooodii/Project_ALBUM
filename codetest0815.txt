// 이름이 없는 동물의 ANIMAL_ID 출력하기 - IS NULL
SELECT ANIMAL_ID
FROM ANIMAL_INS
WHERE NAME IS NULL

SELECT ANIMAL_ID
FROM ANIMAL_INS
WHERE NAME IS NOT NULL
ORDER BY ANIMAL_ID ASC

// Null인 데이터 표기 처리하기 - IFNULL 활용
// IFNULL을 띄어쓰기 처리하면 안됨 
SELECT ANIMAL_TYPE, IFNULL(NAME, 'No name') AS NAME, SEX_UPON_INTAKE
FROM ANIMAL_INS
ORDER BY ANIMAL_ID ASC;

// 없어진 차트 기록 찾기 - LEFT JOIN, LEFT OUTER JOIN 
// LEFT JOIN - 왼쪽 테이블을 기준으로 오른쪽 테이블을 매치, 일치하는 경우 표시 
// LEFT OUTER JOIN - 왼쪽 테이블의 모든 데이터 + 왼/오른쪽 테이블의 중복데이터 검색 

SELECT O.ANIMAL_ID, O.NAME
FROM ANIMAL_OUTS AS O 
LEFT OUTER JOIN ANIMAL_INS AS I
ON O.ANIMAL_ID = I.ANIMAL_ID
WHERE I.ANIMAL_ID IS NULL;
