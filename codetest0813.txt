// ANIMAL_INS 테이블에 들어있는 모든 정보를 ANIMAL_ID순으로 조회
SELECT *
FROM ANIMAL_INS
ORDER BY ANIMAL_ID;

//ANIMAL_INS 테이블에 들어있는 이름, 보호시작일을 ANIMAL_ID 역순으로 출력
SELECT NAME, DATETIME
FROM ANIMAL_INS
ORDER BY ANIMAL_ID DESC;

// ANIMAL_INS 테이블에서 아픈 동물의 아이디와 이름을 아이디 기준으로 출력
SELECT ANIMAL_INS.ANIMAL_ID, ANIMAL_INS.NAME 
FROM ANIMAL_INS
WHERE ANIMAL_INS.INTAKE_CONDITION ='Sick'
ORDER BY ANIMAL_ID;
